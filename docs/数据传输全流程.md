#### **1.2 前端发送 HTTP 请求到后端**
```typescript
// frontend/src/services/api.ts
import axios from 'axios';

export async function generateImage(params: {
  prompt: string;
  model: string;
  imageSize: string;
}) {
  const response = await axios.post(
    'http://localhost:8000/api/generate/image',  // 后端 API 地址
    {
      model: params.model,
      messages: [
        { role: 'user', content: params.prompt }
      ],
      image_config: {
        image_size: params.imageSize
      },
      n: 1
    },
    {
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${localStorage.getItem('token')}`  // 用户 Token
      },
      timeout: 300000  // 5 分钟超时
    }
  );
  
  return response.data;
}
```

**发送的 HTTP 请求：**
```http
POST /api/generate/image HTTP/1.1
Host: localhost:8000
Content-Type: application/json
Authorization: Bearer eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...

{
  "model": "gpt-image-1",
  "messages": [{"role": "user", "content": "一只可爱的猫咪"}],
  "image_config": {"image_size": "1K"},
  "n": 1
}
```

**关键点：**
- ✅ 使用 **Axios**（不是 httpx，因为前端是 JavaScript）
- ✅ 发送 **用户 Token**（JWT）证明身份
- ✅ 请求格式是**业务格式**（符合前端需求）
- ✅ 超时时间设置为 5 分钟（AI 生成可能很慢）

### **阶段 2：后端接收 → 处理 → 调用提供商**

#### **2.1 FastAPI 路由接收请求**
```python
# backend/app/api/routes/generate.py
from fastapi import APIRouter, Depends, HTTPException
from app.services.providers import get_provider
from pydantic import BaseModel

router = APIRouter()

class ImageGenerateRequest(BaseModel):
    model: str
    messages: list[dict]
    image_config: dict | None = None
    n: int = 1

@router.post("/image")
async def generate_image(
    request: ImageGenerateRequest,
    provider = Depends(get_provider)  # 依赖注入获取提供商
):
    try:
        # 验证用户身份（通过 JWT Token）
        # Token 在中间件中已验证
        
        # 调用提供商生成图片
        result = await provider.generate_image(request.dict())
        return result
        
    except ProviderError as e:
        raise HTTPException(status_code=e.status_code, detail=e.message)
```

**后端收到请求后：**
- ✅ **验证用户身份**（JWT Token 已在中间件验证）
- ✅ **解析请求参数**（Pydantic 模型验证）
- ✅ **获取配置的提供商**（OpenAI 或 OpenRouter）
- ✅ **调用提供商的 generate_image 方法**

#### **2.2 提供商转换请求格式**
```python
# backend/app/services/providers/openai_provider.py
async def generate_image(self, payload: dict[str, Any]) -> dict[str, Any]:
    # 1. 检查 API Key 是否配置
    self._require_api_key()
    
    # 2. 从业务格式提取参数
    prompt = payload.get("messages", [{}])[0].get("content", "")
    model = payload.get("model", "gpt-image-1")
    image_size = payload.get("image_config", {}).get("image_size", "1K")
    
    # 3. 转换为 OpenAI API 格式
    size_map = {
        "1K": "1024x1024",
        "2K": "1536x1536",
        "4K": "2048x2048",
    }
    
    request_payload = {
        "model": model,
        "prompt": prompt,
        "size": size_map.get(image_size, "1024x1024"),
        "n": max(1, min(4, int(payload.get("n", 1))))
    }
    
    # 4. 调用 HTTP 工具函数
    data = await self._post_json(
        "/images/generations",  # OpenAI API 端点
        request_payload,        # 转换后的格式
        timeout=120.0
    )
    
    # 5. 转换响应格式（稍后处理）
    return self._format_response(data)
```

**格式转换过程：**
```
前端发送:
{
  "messages": [{"role": "user", "content": "一只可爱的猫咪"}],
  "image_config": {"image_size": "1K"}
}
    ↓ 提供商转换 ↓
OpenAI 格式:
{
  "prompt": "一只可爱的猫咪",
  "size": "1024x1024"
}
```

#### **2.3 HTTP 工具函数发送请求**
```python
# backend/app/services/providers/openai_provider.py
async def _post_json(
    self, path: str, payload: dict[str, Any], timeout: float
) -> dict[str, Any]:
    return await provider_post_json(
        provider_name="openai",
        url=f"{self.settings.openai_base_url}{path}",
        headers=self._headers,  # 包含 API Key
        payload=payload,
        timeout=timeout,
    )
```

```python
# backend/app/services/providers/base.py
async def provider_post_json(
    *,
    provider_name: str,
    url: str,
    headers: dict[str, str],
    payload: dict[str, Any],
    timeout: float,
) -> dict[str, Any]:
    try:
        async with httpx.AsyncClient(timeout=timeout) as client:
            response = await client.post(
                url, 
                headers=headers, 
                json=payload  # 自动序列化为 JSON
            )
            response.raise_for_status()  # 检查状态码
            return response.json()  # 解析 JSON 响应
    except httpx.TimeoutException as exc:
        raise ProviderUpstreamError(
            provider_name,
            code="timeout",
            message="The upstream provider timed out.",
        ) from exc
    # ... 其他错误处理
```

**发送到 OpenAI 的请求：**
```http
POST /v1/images/generations HTTP/1.1
Host: api.openai.com
Authorization: Bearer sk-xxxxxxxxxxxxx  # API Key
Content-Type: application/json

{
  "model": "gpt-image-1",
  "prompt": "一只可爱的猫咪",
  "size": "1024x1024",
  "n": 1
}
```

**关键点：**
- ✅ 使用 **httpx**（Python 异步 HTTP 客户端）
- ✅ 发送 **API Key**（证明付费资格）
- ✅ 请求格式是 **OpenAI API 格式**
- ✅ 超时时间设置为 120 秒

    # ... 前面的代码
    
    data = await self._post_json("/images/generations", request_payload, timeout=120.0)

                    "images": images,
                }
            }
        ],

    }

## 📊 关键对比总结

| 层级 | 通信协议 | 数据格式 | 认证方式 | 工具 | 超时 |
|------|---------|---------|---------|------|------|
| **前端 → 后端** | HTTP/HTTPS | JSON | JWT Token | Axios | 300 秒 |
| **后端 → 提供商** | HTTP/HTTPS | JSON | API Key | httpx | 120 秒 |

| 层级 | 请求格式 | 响应格式 | 数据位置 | 目的 |
|------|---------|---------|---------|------|
| **前端 → 后端** | 业务格式 | 业务格式 | 请求体 | 获取服务 |
| **后端 → 提供商** | API 格式 | API 格式 | 请求体 | 调用 AI |

---

## 🎯 核心要点

1. **前后端分离**：前端和后端独立部署，通过 HTTP 通信
2. **双层转换**：
   - 前端格式 → 提供商格式（请求时）
   - API 格式 → 前端格式（响应时）
3. **认证分离**：
   - 前端用用户 Token（身份）
   - 后端用 API Key（付费）
4. **工具不同**：
   - 前端用 Axios（JavaScript）
   - 后端用 httpx（Python）
5. **统一抽象**：提供商接口隐藏了底层 AI 服务差异

这就是完整的全流程！🎉
